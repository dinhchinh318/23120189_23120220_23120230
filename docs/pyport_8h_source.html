298 /* Py_ARITHMETIC_RIGHT_SHIFT
299  * C doesn't define whether a right-shift of a signed integer sign-extends
300  * or zero-fills. Here a macro to force sign extension:
301  * Py_ARITHMETIC_RIGHT_SHIFT(TYPE, I, J)
302  * Return I >> J, forcing sign extension. Arithmetically, return the
303  * floor of I/2**J.
304  * Requirements:
305  * I should have signed integer type. In the terminology of C99, this can
306  * be either one of the five standard signed integer types (signed char,
307  * short, int, long, long long) or an extended signed integer type.
308  * J is an integer >= 0 and strictly less than the number of bits in the
309  * type of I (because C doesn't define what happens for J outside that
310  * range either).
311  * TYPE used to specify the type of I, but is now ignored. It's been left
312  * in for backwards compatibility with versions <= 2.6 or 3.0.
313  * Caution:
314  * I may be evaluated more than once.
315  */
316 #ifdef SIGNED_RIGHT_SHIFT_ZERO_FILLS
317 #define Py_ARITHMETIC_RIGHT_SHIFT(TYPE, I, J) \
318  ((I) < 0 ? -1-((-1-(I)) >> (J)) : (I) >> (J))
319 #else
320 #define Py_ARITHMETIC_RIGHT_SHIFT(TYPE, I, J) ((I) >> (J))
321 #endif

415 // Ask the compiler to always inline a static inline function. The compiler can
416 // ignore it and decides to not inline the function.
417 //
418 // It can be used to inline performance critical static inline functions when
419 // building Python in debug mode with function inlining disabled. For example,
420 // MSC disables function inlining when building in debug mode.
421 //
422 // Marking blindly a static inline function with Py_ALWAYS_INLINE can result in
423 // worse performances (due to increased code size for example). The compiler is
424 // usually smarter than the developer for the cost/benefit analysis.
425 //
426 // If Python is built in debug mode (if the Py_DEBUG macro is defined), the
427 // Py_ALWAYS_INLINE macro does nothing.
428 //
429 // It must be specified before the function return type. Usage:
430 //
431 // static inline Py_ALWAYS_INLINE int random(void) { return 4; }
432 #if defined(Py_DEBUG)
433  // If Python is built in debug mode, usually compiler optimizations are
434  // disabled. In this case, Py_ALWAYS_INLINE can increase a lot the stack
435  // memory usage. For example, forcing inlining using gcc -O0 increases the
436  // stack usage from 6 KB to 15 KB per Python function call.
437 # define Py_ALWAYS_INLINE
438 #elif defined(__GNUC__) || defined(__clang__) || defined(__INTEL_COMPILER)
439 # define Py_ALWAYS_INLINE __attribute__((always_inline))
440 #elif defined(_MSC_VER)
441 # define Py_ALWAYS_INLINE __forceinline
442 #else
443 # define Py_ALWAYS_INLINE
444 #endif
