188 # define DSS_prime_checks 64
189 /*
190  * Primality test according to FIPS PUB 186-4, Appendix C.3. Since we only
191  * have one value here we set the number of checks to 64 which is the 128 bit
192  * security level that is the highest level and valid for creating a 3072 bit
193  * DSA key.
194  */
