228 #define CHANNEL_NONBLOCKING (1<<3) /* Channel is currently in nonblocking
229  * mode. */
230 #define CHANNEL_LINEBUFFERED (1<<4) /* Output to the channel must be

232 #define CHANNEL_UNBUFFERED (1<<5) /* Output to the channel must always
233  * be flushed immediately. */
234 #define BG_FLUSH_SCHEDULED (1<<7) /* A background flush of the queued
235  * output buffers has been
236  * scheduled. */
237 #define CHANNEL_CLOSED (1<<8) /* Channel has been closed. No further
238  * Tcl-level IO on the channel is
239  * allowed. */
240 #define CHANNEL_EOF (1<<9) /* EOF occurred on this channel. This
241  * bit is cleared before every input
242  * operation. */
243 #define CHANNEL_STICKY_EOF (1<<10) /* EOF occurred on this channel
244  * because we saw the input
245  * eofChar. This bit prevents clearing
246  * of the EOF bit before every input
247  * operation. */
248 #define CHANNEL_BLOCKED (1<<11) /* EWOULDBLOCK or EAGAIN occurred on
249  * this channel. This bit is cleared
250  * before every input or output
